<script>
    // **********************************************
    // SIMULATED BACKEND (in-memory database + API)
    // **********************************************
    // Data stores
    let mentorsDB = [];
    let requestsDB = [];

    // Helper: generate unique id
    function uuid() {
        return Date.now() + '-' + Math.random().toString(36).substring(2, 10);
    }

    // ---------- BACKEND API (async simulation) ----------
    const backendAPI = {
        // Mentor endpoints
        getAllMentors: async () => {
            await delay(180);
            return [...mentorsDB];
        },
        addMentor: async (mentorData) => {
            await delay(250);
            if (!mentorData.name || !mentorData.expertise) throw new Error("Name and expertise required");
            const newMentor = {
                id: uuid(),
                name: mentorData.name.trim(),
                expertise: mentorData.expertise.trim(),
                domain: mentorData.domain,
                createdAt: new Date().toISOString()
            };
            mentorsDB.push(newMentor);
            return newMentor;
        },
        // Request endpoints
        getAllRequests: async () => {
            await delay(150);
            return [...requestsDB];
        },
        createRequest: async (mentorId, studentName, studentEmail) => {
            await delay(300);
            const mentorExists = mentorsDB.find(m => m.id === mentorId);
            if (!mentorExists) throw new Error("Mentor not found");
            if (!studentName?.trim() || !studentEmail?.trim()) throw new Error("Student name and email required");
            const emailRegex = /^[^\s@]+@([^\s@.,]+\.)+[^\s@.,]{2,}$/;
            if (!emailRegex.test(studentEmail)) throw new Error("Please provide a valid email address");
            
            // Avoid duplicate pending request from same student & mentor
            const duplicate = requestsDB.find(r => r.mentorId === mentorId && r.studentEmail === studentEmail && r.status === 'pending');
            if (duplicate) throw new Error("You already have a pending request with this mentor");
            
            const newRequest = {
                id: uuid(),
                mentorId: mentorId,
                studentName: studentName.trim(),
                studentEmail: studentEmail.trim(),
                status: 'pending',   // pending, approved, rejected
                requestedAt: new Date().toISOString()
            };
            requestsDB.push(newRequest);
            return newRequest;
        },
        updateRequestStatus: async (requestId, newStatus) => {
            await delay(200);
            const reqIndex = requestsDB.findIndex(r => r.id === requestId);
            if (reqIndex === -1) throw new Error("Request not found");
            if (!['approved', 'rejected'].includes(newStatus)) throw new Error("Invalid status");
            requestsDB[reqIndex].status = newStatus;
            return requestsDB[reqIndex];
        }
    };

    function delay(ms) { return new Promise(resolve => setTimeout(resolve, ms)); }

    // -------------------------------
    // FRONTEND CONTROLLER (UI + Backend binding)
    // -------------------------------
    const mentorsContainer = document.getElementById('mentorsListContainer');
    const requestsContainer = document.getElementById('requestsListContainer');
    const addMentorBtn = document.getElementById('createMentorBtn');
    const mentorNameInput = document.getElementById('mentorNameInput');
    const mentorExpertiseInput = document.getElementById('mentorExpertiseInput');
    const mentorDomainSelect = document.getElementById('mentorDomainSelect');
    const toastDiv = document.getElementById('globalToast');

    let toastTimer = null;
    function showMessage(msg, isError = false) {
        toastDiv.textContent = msg || (isError ? "⚠️ Operation failed" : "✅ Success");
        toastDiv.style.opacity = '1';
        if (toastTimer) clearTimeout(toastTimer);
        toastTimer = setTimeout(() => {
            toastDiv.style.opacity = '0';
        }, 2600);
    }

    // Load mentors and render
    async function loadMentors() {
        try {
            const mentors = await backendAPI.getAllMentors();
            mentorsDB = mentors;  // sync local
            renderMentors();
        } catch (err) {
            showMessage("Failed to load mentors: " + err.message, true);
        }
    }

    // Load requests and render
    async function loadRequests() {
        try {
            const requests = await backendAPI.getAllRequests();
            requestsDB = requests;
            renderRequests();
        } catch (err) {
            showMessage("Failed to load requests", true);
        }
    }

    // Render mentor cards with Request button
    function renderMentors() {
        if (!mentorsContainer) return;
        if (mentorsDB.length === 0) {
            mentorsContainer.innerHTML = `<div class="empty-state"><i class="fas fa-chalkboard"></i> No mentors yet — become a mentor!</div>`;
            return;
        }
        mentorsContainer.innerHTML = mentorsDB.map(mentor => `
            <div class="mentor-card" data-mentor-id="${mentor.id}">
                <div class="mentor-header">
                    <span class="mentor-name"><i class="fas fa-star-of-life" style="font-size:0.8rem;"></i> ${escapeHtml(mentor.name)}</span>
                    <span class="expertise-tag">${escapeHtml(mentor.expertise)}</span>
                </div>
                <div class="mentor-meta">
                    <span class="badge-domain"><i class="fas fa-tag"></i> ${escapeHtml(mentor.domain)}</span>
                    <span class="badge-domain"><i class="far fa-calendar-alt"></i> Joined: ${new Date(mentor.createdAt).toLocaleDateString()}</span>
                </div>
                <button class="request-btn request-mentor-action" data-id="${mentor.id}" data-name="${escapeHtml(mentor.name)}">
                    <i class="fas fa-handshake"></i> Request Mentorship
                </button>
            </div>
        `).join('');

        // Attach event listeners to request buttons
        document.querySelectorAll('.request-mentor-action').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const mentorId = btn.getAttribute('data-id');
                const mentorName = btn.getAttribute('data-name');
                openRequestDialog(mentorId, mentorName);
            });
        });
    }

    // Custom dialog for mentorship request (simple & elegant)
    function openRequestDialog(mentorId, mentorName) {
        // create modal-like prompt but using window confirm + prompt? better use lightweight custom dialog using built-in? For clean UX: prompt combo
        const studentName = prompt(`📚 Request mentorship from ${mentorName}\n\nEnter your full name:`);
        if (!studentName) return;
        const studentEmail = prompt(`✉️ Great! Please provide your email address for the mentor to contact you:\n(example: name@domain.com)`);
        if (!studentEmail) return;
        // call API to submit request
        submitMentorshipRequest(mentorId, studentName, studentEmail);
    }

    async function submitMentorshipRequest(mentorId, studentName, studentEmail) {
        try {
            const newReq = await backendAPI.createRequest(mentorId, studentName, studentEmail);
            showMessage(`✅ Request sent to mentor! (ID: ${newReq.id.slice(-6)})`);
            await loadRequests();  // refresh requests panel instantly
        } catch (err) {
            showMessage(err.message || "Request failed, try again", true);
        }
    }

    // Render requests with approve/reject functionality (mentor actions)
    function renderRequests() {
        if (!requestsContainer) return;
        if (requestsDB.length === 0) {
            requestsContainer.innerHTML = `<div class="empty-state"><i class="fas fa-comment-dots"></i> No mentorship requests yet. Connect with a mentor!</div>`;
            return;
        }

        // Enrich requests with mentor name
        const enrichedRequests = requestsDB.map(req => {
            const mentor = mentorsDB.find(m => m.id === req.mentorId);
            return {
                ...req,
                mentorName: mentor ? mentor.name : "Unknown Mentor",
                mentorExpertise: mentor ? mentor.expertise : ""
            };
        });

        requestsContainer.innerHTML = enrichedRequests.map(req => `
            <div class="request-card ${req.status}" data-req-id="${req.id}">
                <div class="req-mentor-name">
                    <i class="fas fa-chalkboard-user"></i> <strong>${escapeHtml(req.mentorName)}</strong>
                    <span style="font-size:0.7rem;">(${escapeHtml(req.mentorExpertise)})</span>
                </div>
                <div class="req-student">
                    <i class="fas fa-user-graduate"></i> ${escapeHtml(req.studentName)} · ${escapeHtml(req.studentEmail)}
                </div>
                <div>
                    <span class="status-badge status-${req.status}">
                        ${req.status === 'pending' ? '⏳ Pending' : (req.status === 'approved' ? '✅ Approved' : '❌ Rejected')}
                    </span>
                </div>
                ${req.status === 'pending' ? `
                    <div class="action-buttons">
                        <button class="btn-approve" data-id="${req.id}"><i class="fas fa-check-circle"></i> Approve</button>
                        <button class="btn-reject" data-id="${req.id}"><i class="fas fa-times-circle"></i> Decline</button>
                    </div>
                ` : (req.status === 'approved' ? `<div style="margin-top:0.6rem; font-size:0.7rem; color:#0f6e46;"><i class="fas fa-check-double"></i> Mentorship confirmed — connect via email</div>` : `<div style="margin-top:0.6rem; font-size:0.7rem; color:#b91c1c;"><i class="fas fa-ban"></i> Request declined</div>`)}
            </div>
        `).join('');

        // attach approve and reject handlers
        document.querySelectorAll('.btn-approve').forEach(btn => {
            btn.addEventListener('click', async (e) => {
                const reqId = btn.getAttribute('data-id');
                await updateRequest(reqId, 'approved');
            });
        });
        document.querySelectorAll('.btn-reject').forEach(btn => {
            btn.addEventListener('click', async (e) => {
                const reqId = btn.getAttribute('data-id');
                await updateRequest(reqId, 'rejected');
            });
        });
    }

    async function updateRequest(requestId, newStatus) {
        try {
            const updated = await backendAPI.updateRequestStatus(requestId, newStatus);
            showMessage(`Request ${newStatus === 'approved' ? 'approved ✓' : 'rejected'} successfully`);
            await loadRequests();  // re-render request list
        } catch (err) {
            showMessage(err.message, true);
        }
    }

    // Add new mentor via frontend -> backend
    async function addNewMentor() {
        const name = mentorNameInput.value.trim();
        const expertise = mentorExpertiseInput.value.trim();
        const domain = mentorDomainSelect.value;
        if (!name || !expertise) {
            showMessage("Please enter mentor name and expertise", true);
            return;
        }
        try {
            const added = await backendAPI.addMentor({ name, expertise, domain });
            showMessage(`✨ Mentor "${added.name}" added successfully!`);
            mentorNameInput.value = '';
            mentorExpertiseInput.value = '';
            mentorDomainSelect.value = 'Technology';
            await loadMentors();   // refresh mentor list
            // requests are independent, but no need to reload them
        } catch (err) {
            showMessage(err.message, true);
        }
    }

    // Helper escape HTML
    function escapeHtml(str) {
        if (!str) return '';
        return str.replace(/[&<>]/g, function(m) {
            if (m === '&') return '&amp;';
            if (m === '<') return '&lt;';
            if (m === '>') return '&gt;';
            return m;
        });
    }

    // ---------- SEED DATA: to showcase working application ----------
    async function seedDemoData() {
        // Only seed if mentorsDB is empty
        if (mentorsDB.length === 0) {
            const demoMentors = [
                { name: "Prof. Amelia Hart", expertise: "Full Stack & Cloud", domain: "Technology" },
                { name: "Michael Okonkwo", expertise: "Venture Capital & Strategy", domain: "Business" },
                { name: "Elena Rossi", expertise: "UI/UX & Product Design", domain: "Design" },
                { name: "Dr. James Liu", expertise: "Machine Learning Engineering", domain: "Data Science" }
            ];
            for (const mentor of demoMentors) {
                await backendAPI.addMentor(mentor);
            }
            // after seeding, create some demo requests for interactive preview
            const currentMentors = await backendAPI.getAllMentors();
            if (currentMentors.length >= 2) {
                try {
                    await backendAPI.createRequest(currentMentors[0].id, "Olivia Chen", "olivia.chen@student.edu");
                    await backendAPI.createRequest(currentMentors[1].id, "Marcus Wright", "marcus.w@techstart.org");
                    await backendAPI.createRequest(currentMentors[2].id, "Sophia Ramirez", "sophia.r@designlab.com");
                } catch(e) { console.log("seed requests note:", e);}
            }
        }
        await loadMentors();
        await loadRequests();
    }

    // Event binding
    addMentorBtn.addEventListener('click', addNewMentor);

    // Initialize full-stack app
    (async function init() {
        await seedDemoData();
    })();
</script>
